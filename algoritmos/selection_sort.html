Selection Sort

Selection Sort iterativo: 

#include <stdio.h>
#include <stdlib.h>

// Função que inverte os valores numéricos de duas variáveis
void swap(int* a, int* b);

// Selection sort iterativo
void selection_sort(int* array, int array_len);

void swap(int* a, int* b){
    int aux = *a;
    *a = *b;
    *b = aux;
}


void selection_sort(int* array, int array_len){
    for (int i = 0; i < array_len - 1; i++){
        int min_index = i;
        for (int j = i + 1; j < array_len; j++){
            if (array[j] < array[min_index])
                min_index = j;
        }
        if (min_index != i)
            swap(&array[i], &array[min_index]); 
    }
}


// Função de teste
int main(void){
    int array_len;
    printf("Tamanho do array: ");
    scanf("%d", &array_len);

    int* array = (int*) malloc((unsigned) array_len * sizeof(int));
    int input;
    printf("\n");
    for (int i = 0; i < array_len; i++){
        scanf("%d", &input);
        array[i] = input;
    }

    selection_sort(array, array_len);
    printf("\n");
    for (int i = 0; i < array_len; i++){
        printf("%d ", array[i]);
    }
    printf("\n");
    free(array);

    return 0;
}
