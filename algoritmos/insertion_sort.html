Insertion Sort

Insertion Sort iterativo: 

#include <stdio.h>
#include <stdlib.h>

// Função que inverte os valores numéricos de duas variáveis
void swap(int* a, int* b);

// Insertion sort iterativo
void insertion_sort(int* array, int array_len);

void swap(int* a, int* b){
    int aux = *a;
    *a = *b;
    *b = aux;
}


void insertion_sort(int* array, int array_len){
    for (int i = 1; i < array_len; i++){
        int key = array[i];
        int j = i - 1;
        while (j >= 0 && array[j] > key){
            array[j + 1] = array[j];
            j--;
        }
        array[j + 1] = key; 
    }
}


// Função de teste
int main(void){
    int array_len;
    printf("Tamanho do array: ");
    scanf("%d", &array_len);

    int* array = (int*) malloc((unsigned) array_len * sizeof(int));
    int input;
    printf("\n");
    for (int i = 0; i < array_len; i++){
        scanf("%d", &input);
        array[i] = input;
    }

    insertion_sort(array, array_len);
    printf("\n");
    for (int i = 0; i < array_len; i++){
        printf("%d ", array[i]);
    }
    printf("\n");
    free(array);

    return 0;
}
