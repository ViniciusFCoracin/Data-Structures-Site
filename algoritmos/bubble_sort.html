Bubble Sort

Bubble Sort iterativo: 

#include <stdio.h>
#include <stdlib.h>

// Função que inverte os valores numéricos de duas variáveis
void swap(int* a, int* b);

// Bubble sort iterativo
void bubble_sort(int* array, int array_len);

void swap(int* a, int* b){
    int aux = *a;
    *a = *b;
    *b = aux;
}


void bubble_sort(int* array, int array_len){
    for (int i = 0; i < array_len - 1; i++){
        for (int j = 0; j < array_len - 1 - i; j++){
            if (array[j] > array[j + 1])
                swap(&array[j], &array[j + 1]);
        }
    }
}

// Função de teste
int main(void){

Bubble Sort otimizado 

// Bubble sort iterativo otimizado
void optim_bubble_sort(int* array, int array_len);

void optim_bubble_sort(int* array, int array_len){
    for (int i = 0; i < array_len - 1; i++){
        int changed = 0;
        for (int j = 0; j < array_len - 1 - i; j++){
            if (array[j] > array[j + 1]){
                swap(&array[j], &array[j + 1]);
                changed = 1;
            }
        }
        if (!changed) return;
    }
}

Bubble Sort recursivo 

// Bubble sort recursivo. n é o número de iterações realizadas a cada
// chamada de função. Na primeira chamada, n = array_len - 1
void bubble_sort_rec(int* array, int n);

void bubble_sort_rec(int* array, int n){
    if (n == 0) return;

    for (int i = 0; i < n; i++){
        if (array[i] > array[i + 1])
            swap(&array[i], &array[i + 1]);
    }
    bubble_sort_rec(array, n - 1);
}
